{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running:
        kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "avapigw.fullname" . }}

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "avapigw.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "HTTP:    http://$SERVICE_IP:{{ .Values.service.httpPort }}"
  echo "Metrics: http://$SERVICE_IP:{{ .Values.service.metricsPort }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "avapigw.fullname" . }} {{ .Values.service.httpPort }}:{{ .Values.service.httpPort }} &
  echo "HTTP:    http://127.0.0.1:{{ .Values.service.httpPort }}"
  
  # For metrics endpoint:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "avapigw.fullname" . }} {{ .Values.service.metricsPort }}:{{ .Values.service.metricsPort }} &
  echo "Metrics: http://127.0.0.1:{{ .Values.service.metricsPort }}/metrics"
  echo "Health:  http://127.0.0.1:{{ .Values.service.metricsPort }}/health"
{{- end }}
